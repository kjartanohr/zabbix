1 #!/bin/perl
 2 BEGIN{push @INC,"/usr/share/zabbix/bin/perl-5.10.1/lib"}
 3 
 4 use warnings;
 5 use strict;
 6 
 7 $0 = "perl cleanup files";
 8 $|++;
 9 
10 if (defined $ARGV[0] && $ARGV[0] eq "--zabbix-test-run"){
11   print "ZABBIX TEST OK";
12   exit;
13 }
14 
15 
16 my $debug  = 0;         #This needs to be 0 when running in production 
17 my $size   = "100M";    #File have to be bigger than 
18 
19 my @ignore = (
20   "/home/",             #Debug files
21   '20\d\d-\d\d-\d\d',   #Local log files
22   'fw\.log',            #Local log file
23 );
24 
25 my @search = (
26   "*.elg",
27   "*.log",
28 );
29 
30 
31 foreach my $search (@search){
32 
33   FILE: foreach my $file (`find / -iname "$search" -size +$size 2>/dev/null`){
34     chomp $file;
35 
36     foreach my $ignore (@ignore) {
37       if ($file =~ m#$ignore#){
38         print "Ignoring $file. $file =~ /$ignore/\n" if $debug;
39         next FILE;
40       }
41     }
42 
43     if ($debug) {
44       print "$file\nFile found and ready for deleting. Press Y to continue: ";
45       my $answer = <>;
46       chomp $answer;
47 
48       next FILE unless $answer eq "Y";
49     }
50 
51     print "$file\n"; #This is return data to zabbix for the trigger.
52     unlink $file;
53   }
54 }
